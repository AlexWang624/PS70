Assignment: Make a Capacitive Sensor

Idea:

I made a bend sensor with velostat and conductive tape that the resistace reduces when squeezed.


Materials:

1. Velostat

2. Conductive Tape

1. Paper Tape

2. Jumper Wires


Code:

      
        // Define pin numbers
        const int trigPin = 9;
        const int echoPin = 10;
        // Define LED pins
        const int ledPins[] = {2, 3, 4, 5, 6};
        const int numLeds = sizeof(ledPins) / sizeof(ledPins[0]);

        void setup() {
        // Initialize serial communication
        Serial.begin(9600);
        // Define sensor pins
        pinMode(trigPin, OUTPUT);
        pinMode(echoPin, INPUT);
        // Initialize LED pins
        for (int i = 0; i < numLeds; i++) { pinMode(ledPins[i], OUTPUT); } } void loop() { // Clear the trigPin
          digitalWrite(trigPin, LOW);
          delayMicroseconds(2);
          // Set the trigPin high for 10 microseconds
          digitalWrite(trigPin, HIGH);
          delayMicroseconds(10);
          digitalWrite(trigPin, LOW);
          // Read the echoPin and calculate the distance
          long duration=pulseIn(echoPin, HIGH);
          long distance=duration * 0.034 / 2; // Speed of sound wave divided by 2 (go and back)
          Serial.println(distance);

          // Control LEDs based on distance
          for (int i=0; i < numLeds; i++) { if (distance> (i + 1) * 4) { // Adjust the multiplication factor as needed
            digitalWrite(ledPins[i], HIGH);
            } else {
            digitalWrite(ledPins[i], LOW);
            }
            }

            delay(100); // Delay a bit to avoid spamming
            }
